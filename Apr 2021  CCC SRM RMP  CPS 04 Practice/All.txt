//I M17 - Reverse Vowels

#include <stdio.h>
#include<string.h>
int main() {
    char line[150];
    float vowels=0,len;
    fgets(line, sizeof(line), stdin);
    len=strlen(line);
    

    for (int i = 0; line[i] != '\0'; ++i) {
        if (line[i] == 'a' || line[i] == 'e' || line[i] == 'i' ||
            line[i] == 'o' || line[i] == 'u' || line[i] == 'A' ||
            line[i] == 'E' || line[i] == 'I' || line[i] == 'O' ||
            line[i] == 'U') {
            ++vowels;
        }
    }

    printf("%.4f",(vowels/len)*100);
    return 0;
}
